Add a new `coin_sort.initTutorial(board_spec)` function after `coin_sort.init()` (around line 362). This function sets up a deterministic board for the tutorial without random dealing:

```lua
function coin_sort.initTutorial(board_spec)
    -- Load historical max coin (same as init)
    local prog = require("progression")
    local d = prog.getUpgradesData()
    max_coin_reached = d.max_coin_reached or 0

    -- Use exactly the specified board
    boxes = {}
    active_box_count = board_spec.active_box_count or 4
    for grid_idx, coins in pairs(board_spec.boxes) do
        boxes[grid_idx] = {}
        for _, coin_val in ipairs(coins) do
            table.insert(boxes[grid_idx], {number = coin_val})
        end
    end

    points = 0
    merge_timer = 0
    error_timer = 0
    error_message = ""
    total_merges = 0
    max_spawn_number = board_spec.max_spawn_number or 2
    game_active = true

    commissions.init()
end
```

8. Add the `drawSpotlightOverlay()` local function to coin_sort_screen.lua (before the draw function). This is the ONLY place that calls love.graphics for the spotlight. It reads all data from tutorial.lua:

```lua
local function drawSpotlightOverlay()
    local rect = tutorial.getSpotlight()
    if not rect then
        -- Step with no spotlight (e.g., free play) -- just draw text
        local text = tutorial.getText()
        if text ~= "" then
            love.graphics.setColor(0.92, 0.88, 0.78, 1)
            love.graphics.printf(text, 40, VH / 2 - 20, VW - 80, "center")
        end
        return
    end

    local opacity = tutorial.getOverlayOpacity()
    local r = tutorial.getCutoutRadius()

    -- Phase 1: write cutout into stencil buffer
    love.graphics.setStencilMode("draw", 1)
    love.graphics.setColor(1, 1, 1, 1)
    love.graphics.rectangle("fill", rect.x, rect.y, rect.w, rect.h, r, r)

    -- Phase 2: draw dark overlay everywhere except cutout
    love.graphics.setStencilMode("test", 1)
    love.graphics.setColor(0, 0, 0, opacity)
    love.graphics.rectangle("fill", 0, 0, VW, VH)

    -- Phase 3: reset stencil
    love.graphics.setStencilMode()
    love.graphics.setColor(1, 1, 1, 1)

    -- Phase 4: pulse border (D-03)
    local pulse = tutorial.getPulseAlpha()
    love.graphics.setColor(1, 1, 0.4, pulse)
    love.graphics.setLineWidth(3)
    love.graphics.rectangle("line", rect.x, rect.y, rect.w, rect.h, r, r)
    love.graphics.setLineWidth(1)

    -- Phase 5: hand icon animation (D-13, D-14)
    local hand = tutorial.getHandAnim()
    if hand.type == "tap" then
        local cx = rect.x + rect.w / 2
        local cy = rect.y + rect.h / 2
        -- Tap motion: hand moves down 20px and back
        local offset_y = 0
        if hand.progress < 0.3 then
            offset_y = (hand.progress / 0.3) * 20
        elseif hand.progress < 0.5 then
            offset_y = 20
        else
            offset_y = 20 * (1 - (hand.progress - 0.5) / 0.5)
        end
        -- Draw pointing hand (triangle + circle, stylized)
        love.graphics.setColor(1, 1, 1, 0.9)
        local hx, hy = cx + 10, cy + offset_y + 15
        -- Simple hand shape: filled triangle pointing down-left
        love.graphics.polygon("fill", hx, hy, hx - 16, hy - 28, hx + 16, hy - 28)
        love.graphics.circle("fill", hx, hy - 28, 10)
    elseif hand.type == "drag" and hand.drag_target then
        local sx = rect.x + rect.w / 2
        local sy = rect.y + rect.h / 2
        local tx, ty = hand.drag_target.x, hand.drag_target.y
        -- Drag trail: hand moves from source to target
        local prog = math.min(hand.progress / 0.8, 1)
        local hx = sx + (tx - sx) * prog
        local hy = sy + (ty - sy) * prog
        -- Draw trail line
        love.graphics.setColor(1, 1, 0.4, 0.3)
        love.graphics.setLineWidth(4)
        love.graphics.line(sx, sy, hx, hy)
        love.graphics.setLineWidth(1)
        -- Draw hand at current position
        love.graphics.setColor(1, 1, 1, 0.9)
        love.graphics.polygon("fill", hx, hy, hx - 16, hy - 28, hx + 16, hy - 28)
        love.graphics.circle("fill", hx, hy - 28, 10)
    end

    -- Phase 6: instruction text (D-15, D-16)
    local text = tutorial.getText()
    if text ~= "" then
        local pos = tutorial.getTextPosition(rect)
        local text_y
        if pos == "above" then
            text_y = rect.y - 60
        else
            text_y = rect.y + rect.h + 20
        end
        -- Text background
        love.graphics.setColor(0, 0, 0, 0.7)
        love.graphics.rectangle("fill", 40, text_y - 8, VW - 80, 50, 8, 8)
        -- Text
        love.graphics.setColor(0.92, 0.88, 0.78, 1)
        love.graphics.printf(text, 50, text_y, VW - 100, "center")
    end
end
```
